using System;

class Program
{
    static void Main()
    {
        // Entrada de dados
        Console.Write("Nome do Herói: ");
        string nome = Console.ReadLine() ?? "";

        Console.Write("Classe do Herói (Mago, Clérigo, Guerreiro, Druida, Anão, Elfo ou Troll): ");
        string classe = Console.ReadLine() ?? "";

        Console.Write("Nível: ");
        int nivel;
        int.TryParse(Console.ReadLine(), out nivel);

        Console.Write("Pontos de Vida (HP): ");
        double hp;
        double.TryParse(Console.ReadLine(), out hp);

        Console.Write("Pontos de Magia (Mana): ");
        int mana;
        int.TryParse(Console.ReadLine(), out mana);

        Console.Write("Ouro na mochila: ");
        double ouro;
        double.TryParse(Console.ReadLine(), out ouro);

        Console.WriteLine("\n--- Avaliação da Guilda ---");

        // Regra da raça inimiga
        if (classe == "Troll")
        {
            Console.WriteLine("Troll detectado! Expulso da guilda imediatamente! 🚫");
            return;
        }
        else if (classe != "Troll" && !(nivel < 5))
        {
            Console.WriteLine("Herói aceito na guilda! 🏛️");
        }
        else
        {
            Console.WriteLine("Nível muito baixo para entrar na guilda.");
            return;
        }

        // Cálculo do poder base
        double poderBase = (nivel * 10) + (hp / 2) - 5;
        Console.WriteLine("Poder Base do herói: " + poderBase);

        // Regra do dano crítico
        if (poderBase % 2 == 0)
        {
            Console.WriteLine("⚔️ Ataque Crítico liberado!");
        }

        bool podeAtacar = false;

        // Regra da magia
        if (classe == "Mago" || classe == "Druida" || classe == "Clérigo")
        {
            if (mana >= 30 && nivel <= 100)
            {
                Console.WriteLine("O herói lança um poderoso feitiço contra o dragão! 🧙‍♂️");
                podeAtacar = true;
            }
            else
            {
                Console.WriteLine("Mana insuficiente! O cajado falhou! 🪵");
            }
        }

        // Regra do ataque físico
        else if (classe == "Guerreiro" || classe == "Anão" || classe == "Elfo")
        {
            if (hp > 20 && poderBase > 50)
            {
                Console.WriteLine("O herói parte para o ataque físico! ⚔️");
                podeAtacar = true;
            }
            else
            {
                Console.WriteLine("O herói está fraco demais para atacar.");
            }
        }

        // Suborno ao dragão
        if (!podeAtacar)
        {
            Console.WriteLine("O herói tenta subornar o dragão... 💰");

            if (ouro > 1000 || classe == "Anão")
            {
                Console.WriteLine("O dragão aceita o suborno! O herói escapou! 🐉💰");
            }
            else
            {
                Console.WriteLine("O dragão recusou o suborno... virou churrasco! 🔥");
            }
        }

        Console.WriteLine("\nFim da avaliação da guilda.");
    }
}
